#CTEs common table expression 

with CTE_example AS
(
select gender,avg(salary),max(salary),count(salay)
from employee_demographics dem 
join employee_salary sal
	on dem.employee_id = sal.employee_id
group by gender
)
select *
from cte_example
;

#we can write it with the subquery
with CTE_example AS
(
select gender,avg(salary) avg_sal,
max(salary) max_sal,
min(salary) min_sal,
count(salay) count_sal,
from employee_demographics dem 
join employee_salary sal
	on dem.employee_id = sal.employee_id
group by gender
)
select *
from cte_example
;
